    ```javascript
    export function useBatchProcessor() {
      const isPausedRef = useRef(false)
      const isCancelledRef = useRef(false)
      const [isPaused, setIsPaused] = useState(false)
      const [statuses, setStatuses] = useState({})  // { itemId: 'pending'|'processing'|'done'|'skipped'|errorString }
      const [isRunning, setIsRunning] = useState(false)
      const itemsRef = useRef([])  // store items for retry

      const pause = () => { isPausedRef.current = true; setIsPaused(true) }
      const resume = () => { isPausedRef.current = false; setIsPaused(false) }
      const cancel = () => { isCancelledRef.current = true }

      const run = async (items, concurrency, processFn) => {
        // Reset state
        isCancelledRef.current = false
        isPausedRef.current = false
        setIsPaused(false)
        itemsRef.current = items
        const initial = {}
        for (const item of items) initial[item.id] = 'pending'
        setStatuses(initial)
        setIsRunning(true)

        const queue = [...items]
        const worker = async () => {
          while (queue.length > 0) {
            if (isCancelledRef.current) break
            while (isPausedRef.current && !isCancelledRef.current) {
              await new Promise(r => setTimeout(r, 200))
            }
            if (isCancelledRef.current) break
            const item = queue.shift()
            if (!item) break
            setStatuses(prev => ({ ...prev, [item.id]: 'processing' }))
            try {
              const result = await processFn(item)
              setStatuses(prev => ({ ...prev, [item.id]: result === 'skipped' ? 'skipped' : 'done' }))
            } catch (err) {
              setStatuses(prev => ({ ...prev, [item.id]: err.message || 'Failed' }))
            }
          }
        }

        await Promise.all(Array.from({ length: Math.min(concurrency, items.length) }, worker))

        // Mark remaining pending items as cancelled if cancelled
        if (isCancelledRef.current) {
          setStatuses(prev => {
            const next = { ...prev }
            for (const [id, s] of Object.entries(next)) {
              if (s === 'pending') next[id] = 'cancelled'
            }
            return next
          })
        }

        setIsRunning(false)
      }

      const getFailedItems = () => {
        return itemsRef.current.filter(item => {
          const s = statuses[item.id]
          return s && s !== 'pending' && s !== 'processing' && s !== 'done' && s !== 'skipped' && s !== 'cancelled'
        })
      }

      const retryFailed = async (concurrency, processFn) => {
        const failed = getFailedItems()
        if (failed.length === 0) return
        await run(failed, concurrency, processFn)
      }

      return { run, pause, resume, cancel, isPaused, statuses, setStatuses, isRunning, getFailedItems, retryFailed }
    }
    ```
