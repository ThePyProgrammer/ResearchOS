```js
function buildExperimentTree(flatExperiments) {
  const byId = Object.fromEntries(flatExperiments.map(e => [e.id, { ...e, children: [] }]))
  const roots = []
  for (const exp of Object.values(byId)) {
    if (exp.parentId) byId[exp.parentId]?.children.push(exp)
    else roots.push(exp)
  }
  return roots
}
```
