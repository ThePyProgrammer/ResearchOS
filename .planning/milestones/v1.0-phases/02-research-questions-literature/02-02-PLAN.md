**1. Tree builder utility function (top of file):**
```javascript
function buildRqTree(flatRqs) {
  const byId = Object.fromEntries(flatRqs.map(rq => [rq.id, { ...rq, children: [] }]))
  const roots = []
  for (const rq of Object.values(byId)) {
    if (rq.parentId) byId[rq.parentId]?.children.push(rq)
    else roots.push(rq)
  }
  const sortLevel = nodes => { nodes.sort((a, b) => a.position - b.position); nodes.forEach(n => sortLevel(n.children)) }
  sortLevel(roots)
  return roots
}
```
