```javascript
// Root-onto-root: check if dropping ONTO the target (demote) vs BETWEEN (reorder)
if (draggedParentId === null && targetParentId === null) {
  // Get the target element's bounding rect
  const overElement = document.querySelector(`[data-rq-id="${over.id}"]`)
  if (overElement) {
    const rect = overElement.getBoundingClientRect()
    const pointerY = lastPointerY.current
    const topZone = rect.top + rect.height * 0.25
    const bottomZone = rect.bottom - rect.height * 0.25

    // If pointer is in the middle 50% of the target node, treat as "drop onto" (demote)
    if (pointerY > topZone && pointerY < bottomZone) {
      // Check childless constraint
      if (draggedNode.children && draggedNode.children.length > 0) {
        console.warn('Cannot reparent an RQ that has sub-questions.')
        return
      }
      // Demote: make dragged item a child of target
      const targetChildren = flatTree
        .filter(n => (n._parentId || null) === targetNode.id)
        .sort((a, b) => a.position - b.position)
      const newPosition = targetChildren.length // append at end

      try {
        await researchQuestionsApi.update(draggedNode.id, {
          parent_id: targetNode.id,
          position: newPosition,
        })
        await fetchRqs()
      } catch (err) {
        console.error('Failed to demote research question:', err)
        await fetchRqs()
      }
      return
    }
  }
  // If pointer is in top/bottom 25%, fall through to sibling reorder below
}
```
